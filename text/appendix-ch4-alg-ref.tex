\chapter{\label{app:4:algo-ref}Additional Algorithm Details For Chapter 4}

    This appendix gives the full equations for AR-BTS and AR-DENTS for reference. Each algorithm is specified minimally using the \thtspp\ewe schema given in Section \ref{sec:2-4-2-thts}.






\section{AR-BTS}
\label{appsec:4:ar-bts-ref}

    \paragraph{Value Estimates and Parameters.}
    
        \begin{itemize}
            \item $\Varbts(s)$ - The value estimate for state $s$;
            \item $\Qarbts(s,a)$ - The Q-value estimate for state $s$ and action $a$;
            \item $N(s)$ - The visit count for state $s$;
            \item $N(s,a)$ - The visit count for state $s$ and action $a$;
            \item $\alphaarbts(x)>0$ - The search temperature function;
            \item $\epsarbts \in [0,\infty)$ - The uniform exploration parameter;
            \item $\normalisedq$ - A flag indicating whether the Q-values should be normalised when used in the search policy;
            \item $\Nvarbts$ - The number of pseudo-visits to use for biasing the backups;
            \item $\Nqarbts$ - The number of pseudo-visits to use for biasing the Q-values;
            \item $\Vinit(s)$ - The heuristic value estimate for state $s$;
            \item $\Qinit(s,a)$ - The heuristic Q-value estimate for state $s$ and action $a$;
        \end{itemize}

    \paragraph{Initialisation.}
        \begin{align}
            N(s) \leftarrow 0, \\
            N(s,a) \leftarrow 0, \\
            \Varbts(s) \leftarrow 0, \\
            \Qarbts(s,a) \leftarrow 0.
        \end{align}

    \paragraph{Search Policy.}
        Intermediate Q-values:
        \begin{align}
            \Qarbtsbias(s,a) &= \frac{N(s,a) \Qarbts(s,a) +  \Nqarbts \Qinit(s,a)}{N(s,a)+\Nqarbts}, \\
            \Qbtsnorm(s,a) &= 
            \begin{cases}
                \frac{\Qarbtsbias(s,a)-\min_{a\in\cl{A}}\Qarbtsbias(s,a)}{\max_{a\in\cl{A}}\Qarbtsbias(s,a)-\min_{a\in\cl{A}}\Qarbtsbias(s,a)} & \text{if } \normalisedq, \\
                \Qarbtsbias(s,a) & \text{otherwise.}
            \end{cases}
        \end{align}

        Search policy:
        \begin{align}
            \piarbts(a|s) &= (1-\lambdaarbts)\rhoarbts(a|s) + \frac{\lambdaarbts}{|\cl{A}|}, \\ 
            \rhoarbts(a|s) &\propto \exp\left(\frac{\Qarbtsnorm(s,a)}{\alphaarbts(N(s))}\right), \\
            \lambdaarbts &= \min\left(1, \frac{\epsarbts}{\log(e+N(s))}\right).
        \end{align}

    \paragraph{Backups.}
        Let $\tau_{:h}=(s_0,a_0,r_0,\ldots,s_{h-1},a_{h-1},r_{h-1},s_h)\sim\piarbts$:
        \begin{align}
            \Qarbts(s_t,a_t) &\leftarrow 
                R(s_t,a_t) + \sum_{s' \in \suc{s_t}{a_t}} \left( \frac{N(s')}{N(s_t,a_t)} \Varbts(s') \right),  \\ 
            \Varbts(s_t) &\leftarrow \frac{\Nvarbts \Vinit(s_t)}{N(s_t)+\Nvarbts} 
                + \sum_{a\in\cl{A}} \frac{N(s_t,a)}{N(s_t)+\Nvarbts} \Qarbts(s_t,a).
        \end{align}

    \paragraph{Recommendation Policy.}
        \begin{align}
            \psiarbts(s)=\argmax_{a\in\cl{A}}\Qarbts(s,a),  \\
            \mvarbts(s) = \argmax_{a\in\cl{A}} N(s,a). 
        \end{align}




\section{AR-DENTS}
\label{appsec:4:ar-dents-ref}

    \paragraph{Value Estimates and Parameters.}

        \begin{itemize}
            \item $\Vardents(s)$ - The value estimate for state $s$;
            \item $\Qardents(s,a)$ - The Q-value estimate for state $s$ and action $a$;
            \item $N(s)$ - The visit count for state $s$;
            \item $N(s,a)$ - The visit count for state $s$ and action $a$;
            \item $\alphaardents(x)>0$ - The search temperature function;
            \item $\epsardents \in [0,\infty)$ - The uniform exploration parameter;
            \item $\normalisedq$ - A flag indicating whether the Q-values should be normalised when used in the search policy;
            \item $\Nvardents$ - The number of pseudo-visits to use for biasing the backups;
            \item $\Nqardents$ - The number of pseudo-visits to use for biasing the Q-values;
            \item $\Vinit(s)$ - The heuristic value estimate for state $s$;
            \item $\Qinit(s,a)$ - The heuristic Q-value estimate for state $s$ and action $a$;
        \end{itemize}

    \paragraph{Initialisation.}
        \begin{align}
            N(s) \leftarrow 0, \\
            N(s,a) \leftarrow 0, \\
            \Vardents(s) \leftarrow 0, \\
            \Qardents(s,a) \leftarrow 0.
        \end{align}

    \paragraph{Search Policy.}
        Intermediate Q-values:
        \begin{align}
            \Qardentssoft(s,a) &= \Qardents(s,a)+\betaardents(N(s))\HQardents(s,a), \\
            \Qardentsbias(s,a) &= \frac{N(s,a) \Qardentssoft(s,a) +  \Nqardents \Qinit(s,a)}{N(s,a)+\Nqardents}, \\
            \Qardentsnorm(s,a) &= 
            \begin{cases}
                \frac{\Qardentsbias(s,a)-\min_{a\in\cl{A}}\Qardentsbias(s,a)}{\max_{a\in\cl{A}}\Qardentsbias(s,a)-\min_{a\in\cl{A}}\Qardentsbias(s,a)} & \text{if } \normalisedq, \\
                \Qardentsbias(s,a) & \text{otherwise.}
            \end{cases}
        \end{align}

        Search policy:
        \begin{align}
            \piardents(a|s) &= (1-\lambdaardents)\rhoardents(a|s) + \frac{\lambdaardents}{|\cl{A}|}, \\ 
            \rhoardents(a|s) &\propto \exp\left(\frac{\Qardentsnorm(s,a)}{\alphaardents(N(s))}\right), \\
            \lambdaardents &= \min\left(1, \frac{\epsardents}{\log(e+N(s))}\right).
        \end{align}

    \paragraph{Backups.}
        Let $\tau_{:h}=(s_0,a_0,r_0,\ldots,s_{h-1},a_{h-1},r_{h-1},s_h)\sim\piardents$:
        \begin{align}
            \HQardents(s_t,a_t) &\leftarrow \sum_{s'\in \suc{s_t}{a_t}} \frac{N(s')}{N(s_t,a_t)} \HVardents(s'), \\
            \HVardents(s_t) &\leftarrow \cl{H}(\piardents(\cdot | s_t)) + \sum_{a\in\cl{A}} \piardents(a|s_t)\HQardents(s_t,a), \\
            \Qardents(s_t,a_t) &\leftarrow 
                R(s_t,a_t) + \sum_{s' \in \suc{s_t}{a_t}} \left( \frac{N(s')}{N(s_t,a_t)} \Vardents(s') \right),  \\ 
            \Vardents(s_t) &\leftarrow \frac{\Nvardents \Vinit(s_t)}{N(s_t)+\Nvardents} 
                + \sum_{a\in\cl{A}} \frac{N(s_t,a)}{N(s_t)+\Nvardents} \Qardents(s_t,a).
        \end{align}

    \paragraph{Recommendation Policy.}
        \begin{align}
            \psiardents(s)=\argmax_{a\in\cl{A}}\Qardents(s,a),  \\
            \mvardents(s) = \argmax_{a\in\cl{A}} N(s,a). 
        \end{align}
